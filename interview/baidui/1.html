public class BattleshipCounter { public static int countBattleships(char[][] board) { if (board == null || board.length == 0 || board[0].length == 0) { return 0; } int count = 0; int rows = board.length; int cols = board[0].length; for (int i = 0; i < rows; i++) { for (int j=0; j < cols; j++) { // Check if the current cell is 'X' and is the start of a new battleship if (board[i][j]=='X' && (i==0 || board[i - 1][j] !='X' ) && (j==0 || board[i][j - 1] !='X' )) { count++; } } } return count; } public static void main(String[] args) { char[][] board={ {'X', '.' , '.' , 'X' }, {'.', '.' , '.' , 'X' }, {'.', '.' , '.' , 'X' } }; System.out.println(countBattleships(board)); // Expected output: 2 } }
